" ====== System Detection & Paths ======
if has('win32') || has('win64')
  let g:vim_home = expand('~/vimfiles')
else
  let g:vim_home = expand('~/.vim')
endif

let g:undo_dir = g:vim_home . '/undo//'
let g:plug_dir = g:vim_home . '/plugged'

" Create undo directory if it doesn't exist
if !isdirectory(g:undo_dir)
  call mkdir(g:undo_dir, 'p')
endif


" ====== Core Settings ======
set nocompatible " disable compatibility with VI, which can cause unexpected issues
set encoding=utf-8 " support more characters
set hidden " switch buffers without saving current file first
set confirm " instead of failing a command because of unsaved changes, ask for confirmation
set mouse=a " enable mouse usage (all modes)
set clipboard=unnamedplus " use system clipboard, check with :echo has('clipboard')
set visualbell " blink cursor on error, instead of beeping
set noswapfile " prevents swap files from cluttering directories


" ====== Undo / Backup ======
set undofile
execute 'set undodir=' . g:undo_dir
" set autowrite " automatically save before commands like :next and :make


" ====== UI Settings ======
set number " add numbers to left side of line
" set relativenumber " useful for navigation with counts
set scrolloff=2 " display 2 lines above/below the cursor when scrolling
set laststatus=2 " status bar
" set modelines=0 " turn off modelines
set showmode " show the current mode
set showcmd " show (partial) command in status line
set cursorline " higlight cursor line horizontally
set background=dark
set splitright
set splitbelow


" ====== Editing ======
set autoindent " copy indent from current line when starting new line
set smartindent " even better autoindent (add indent after '{')
set expandtab " use space instead of tab
set tabstop=2 " set number of spaces in tab
set shiftwidth=2 " set shift to 2 spaces
set softtabstop=2 " set tab to 2 spaces
set textwidth=120 " text width for wrapping
set colorcolumn=+1 " column line after text width
set nowrap " warping lines (wrap/nowrap)
set showmatch " show matching brackets
set backspace=indent,eol,start " fix backspace
set noshiftround


" ====== Search ======
set incsearch " incremental search
set ignorecase " do case insensitive matching
set smartcase " do smart case matching
set hlsearch " highlights search results


" ====== Performance ======
set updatetime=300 " faster update
set timeoutlen=1000


" ====== Filetypes & Syntax ======
filetype plugin indent on " enable file-type=specific plugins and indentation
syntax on " enable syntax highlighting
colorscheme nord


" ====== Custom ======
set statusline=%F%m%r%h%w\ [FORMAT=%{&ff}]\ [TYPE=%Y]\ [POS=%l,%v][%p%%]\ [BUFFER=%n]
set list " display different types of white spaces.
set listchars=tab:›\ ,trail:•,extends:#,nbsp:.


" press jk to goto normal mode from insert and visual mode
inoremap jk <Esc>
vnoremap jk <Esc>

" move cursor to next line after end/start of line with arrow and hjkl keys
set whichwrap+=<,>,[,],h,l

" move lines up and down with Alt + j/k
" Normal mode
nnoremap <A-j> :m .+1<CR>==
nnoremap <A-k> :m .-2<CR>==
" Visual mode
vnoremap <A-j> :m '>+1<CR>gv=gv
vnoremap <A-k> :m '<-2<CR>gv=gv
" Insert mode
inoremap <A-j> <Esc>:m .+1<CR>==gi
inoremap <A-k> <Esc>:m .-2<CR>==gi

" move lines up and down with Alt + up/down arrow key (if terminal supports)
" Normal mode
" nnoremap <A-Up> :m .-2<CR>==
" nnoremap <A-Down> :m .+1<CR>==
" Visual mode
" vnoremap <A-Up> :m '<-2<CR>gv=gv
" vnoremap <A-Down> :m '>+1<CR>gv=gv
" Insert mode
" inoremap <A-Up> <Esc>:m .-2<CR>==gi
" inoremap <A-Down> <Esc>:m .+1<CR>==gi

" duplicate lines with Ctrl + d in Normal and Insert mode
" nnoremap <C-d> yyp " conflicts in Normal mode with "half-page scroll down"
inoremap <C-d> <Esc>yypgi

" duplicate line/selection with Shirt + Alt + j/k
" Normal mode
nnoremap <S-A-j> yyp
nnoremap <S-A-k> yyP
" Insert mode
inoremap <S-A-j> <Esc>yypgi
inoremap <S-A-k> <Esc>yyPgi
" Visual mode
vnoremap <S-A-j> y'>p
vnoremap <S-A-k> y'<P

" duplicate line/selection with Shift + Alt + Up/Down arrow key
" Normal mode
" nnoremap <S-A-Up> yyP
" nnoremap <S-A-Down> yyp
" Insert mode
" inoremap <S-A-Up> <Esc>yyPgi
" inoremap <S-A-Down> <Esc>yypgi
" Visual Mode
" vnoremap <S-A-Up> y'<P
" vnoremap <S-A-Down> y'>p


if has("gui_running")
  if has("gui_gtk2")
    set guifont=Inconsolata\ 12
  elseif has("gui_macvim")
    set guifont=Menlo\ Regular:h14
  elseif has("gui_win32")
    set guifont=Consolas:h12
  endif
endif

" ====== Restore cursor position ======
augroup remember_cursor
  autocmd!
  autocmd BufReadPost *
        \ if line ("'\"") > 0 && line ("'\"") <= line ("$") |
        \   execute "normal! g'\"" |
        \ endif
  augroup END


" ====== Netrw ======
let g:netrw_banner = 0
let g:netrw_liststyle = 3
let g:netrw_browse_split = 4
let g:netrw_winsize = 25

" ====== vim-plug auto install ======
if empty(glob(g:vim_home . '/autoload/plug.vim'))
  if has('win32') || has('win64')
    silent execute '!powershell -Command "iwr -useb https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim | ni $env:USERPROFILE\vimfiles\autoload\plug.vim -Force"'
  else
    silent execute '!curl -fLo ' . g:vim_home . '/autoload/plug.vim --create-dirs https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim'
  endif
  autocmd VimEnter * PlugInstall
endif

call plug#begin(g:plug_dir)

Plug 'airblade/vim-gitgutter' " Git signs

call plug#end()
